class Solution:
    """
    @param A: A list of integers
    @return: A boolean
    """
    def canJump(self, A):
        # write your code here
        if not A:
            return False
        
        n = len(A)
        dp = [False] * n

        dp[0] = True

        for i in range(1, n):
            for j in range(i):
                if dp[j] == True and j + A[j] >= i:
                    dp[i] = True
                    break
        return dp[n - 1]
